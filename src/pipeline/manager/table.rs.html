53pub(crate) const PIPELINE_TABLE_PIPELINE_NAME_COLUMN_NAME: &str = "name";
54pub(crate) const PIPELINE_TABLE_PIPELINE_SCHEMA_COLUMN_NAME: &str = "schema";
55const PIPELINE_TABLE_PIPELINE_CONTENT_TYPE_COLUMN_NAME: &str = "content_type";

59/// Pipeline table cache size.
60const PIPELINES_CACHE_SIZE: u64 = 10000;
61/// Pipeline table cache time to live.
62const PIPELINES_CACHE_TTL: Duration = Duration::from_secs(10);
63
64/// PipelineTable is a table that stores the pipeline schema and content.
65/// Every catalog has its own pipeline table.
66pub struct PipelineTable {
67    inserter: InserterRef,
68    statement_executor: StatementExecutorRef,
69    table: TableRef,
70    query_engine: QueryEngineRef,
71    pipelines: Cache<String, Arc<Pipeline>>,
72    original_pipelines: Cache<String, (String, TimestampNanosecond)>,
73}

77    pub fn new(
78        inserter: InserterRef,
79        statement_executor: StatementExecutorRef,

82    ) -> Self {
83        Self {

94                .time_to_live(PIPELINES_CACHE_TTL)
95                .build(),
96        }

214    /// Insert a pipeline into the pipeline table.
215    async fn insert_pipeline_to_pipeline_table(
216        &self,
217        schema: &str,
218        name: &str,
219        content_type: &str,
220        pipeline: &str,
221    ) -> Result<Timestamp> {
222        let now = Timestamp::current_time(TimeUnit::Nanosecond);
223
224        let table_info = self.table.table_info();
225
226        let insert = RowInsertRequest {
227            table_name: PIPELINE_TABLE_NAME.to_string(),
228            rows: Some(Rows {
229                schema: Self::build_insert_column_schemas(),
230                rows: vec![Row {
231                    values: vec![
232                        ValueData::StringValue(name.to_string()).into(),
233                        ValueData::StringValue(schema.to_string()).into(),
234                        ValueData::StringValue(content_type.to_string()).into(),
235                        ValueData::StringValue(pipeline.to_string()).into(),
236                        ValueData::TimestampNanosecondValue(now.value()).into(),
237                    ],
238                }],
239            }),
240        };
241
242        let requests = RowInsertRequests {
243            inserts: vec![insert],
244        };
245
246        let output = self
247            .inserter
248            .handle_row_inserts(
249                requests,
250                Self::query_ctx(&table_info),
251                &self.statement_executor,
252                false,
253            )
254            .await
255            .context(InsertPipelineSnafu)?;
256
257        info!(
258            "Insert pipeline success, name: {:?}, table: {:?}, output: {:?}",
259            name,
260            table_info.full_table_name(),
261            output
262        );
263
264        Ok(now)
265    }

267    /// Get a pipeline by name.
268    /// If the pipeline is not in the cache, it will be get from table and compiled and inserted into the cache.
269    pub async fn get_pipeline(
270        &self,
271        schema: &str,
272        name: &str,
273        version: PipelineVersion,
274    ) -> Result<Arc<Pipeline>> {
275        if let Some(pipeline) = self
276            .pipelines
277            .get(&generate_pipeline_cache_key(schema, name, version))
278        {
279            return Ok(pipeline);
280        }
281
282        let pipeline = self.get_pipeline_str(schema, name, version).await?;
283        let compiled_pipeline = Arc::new(Self::compile_pipeline(&pipeline.0)?);
284
285        self.pipelines.insert(
286            generate_pipeline_cache_key(schema, name, version),
287            compiled_pipeline.clone(),
288        );
289        Ok(compiled_pipeline)
290    }

292    /// Get a original pipeline by name.
293    /// If the pipeline is not in the cache, it will be get from table and compiled and inserted into the cache.
294    pub async fn get_pipeline_str(
295        &self,
296        schema: &str,
297        name: &str,
298        version: PipelineVersion,
299    ) -> Result<(String, TimestampNanosecond)> {
300        if let Some(pipeline) = self
301            .original_pipelines
302            .get(&generate_pipeline_cache_key(schema, name, version))
303        {
304            return Ok(pipeline);
305        }
306        let pipeline = self
307            .find_pipeline(schema, name, version)
308            .await?
309            .context(PipelineNotFoundSnafu { name, version })?;
310        self.original_pipelines.insert(
311            generate_pipeline_cache_key(schema, name, version),
312            pipeline.clone(),
313        );
314        Ok(pipeline)
315    }

317    /// Insert a pipeline into the pipeline table and compile it.
318    /// The compiled pipeline will be inserted into the cache.
319    pub async fn insert_and_compile(
320        &self,
321        schema: &str,
322        name: &str,
323        content_type: &str,
324        pipeline: &str,
325    ) -> Result<PipelineInfo> {
326        let compiled_pipeline = Arc::new(Self::compile_pipeline(pipeline)?);
327        // we will use the version in the future
328        let version = self
329            .insert_pipeline_to_pipeline_table(schema, name, content_type, pipeline)
330            .await?;
331
332        {
333            self.pipelines.insert(
334                generate_pipeline_cache_key(schema, name, None),
335                compiled_pipeline.clone(),
336            );
337            self.pipelines.insert(
338                generate_pipeline_cache_key(schema, name, Some(TimestampNanosecond(version))),
339                compiled_pipeline.clone(),
340            );
341
342            self.original_pipelines.insert(
343                generate_pipeline_cache_key(schema, name, None),
344                (pipeline.to_owned(), TimestampNanosecond(version)),
345            );
346            self.original_pipelines.insert(
347                generate_pipeline_cache_key(schema, name, Some(TimestampNanosecond(version))),
348                (pipeline.to_owned(), TimestampNanosecond(version)),
349            );
350        }
351
352        Ok((version, compiled_pipeline))
353    }

355    pub async fn delete_pipeline(
356        &self,
357        schema: &str,
358        name: &str,
359        version: PipelineVersion,
360    ) -> Result<Option<()>> {
361        // 0. version is ensured at the http api level not None
362        ensure!(
363            version.is_some(),
364            InvalidPipelineVersionSnafu { version: "None" }
365        );
366
367        // 1. check pipeline exist in catalog
368        let pipeline = self.find_pipeline(schema, name, version).await?;
369        if pipeline.is_none() {
370            return Ok(None);
371        }
372
373        // 2. prepare dataframe
374        let dataframe = self
375            .query_engine
376            .read_table(self.table.clone())
377            .context(DataFrameSnafu)?;
378        let DataFrame::DataFusion(dataframe) = dataframe;
379
380        let dataframe = dataframe
381            .filter(prepare_dataframe_conditions(schema, name, version))
382            .context(BuildDfLogicalPlanSnafu)?;
383
384        // 3. prepare dml stmt
385        let table_info = self.table.table_info();
386        let table_name = TableReference::full(
387            table_info.catalog_name.clone(),
388            table_info.schema_name.clone(),
389            table_info.name.clone(),
390        );
391
392        let df_schema = Arc::new(
393            table_info
394                .meta
395                .schema
396                .arrow_schema()
397                .clone()
398                .to_dfschema()
399                .context(BuildDfLogicalPlanSnafu)?,
400        );
401
402        // create dml stmt
403        let stmt = DmlStatement::new(
404            table_name,
405            df_schema,
406            datafusion_expr::WriteOp::Delete,
407            Arc::new(dataframe.into_parts().1),
408        );
409
410        let plan = LogicalPlan::Dml(stmt);
411
412        // 4. execute dml stmt
413        let output = self
414            .query_engine
415            .execute(plan, Self::query_ctx(&table_info))
416            .await
417            .context(ExecuteInternalStatementSnafu)?;
418
419        info!(
420            "Delete pipeline success, name: {:?}, version: {:?}, table: {:?}, output: {:?}",
421            name,
422            version,
423            table_info.full_table_name(),
424            output
425        );
426
427        // remove cache with version and latest
428        self.pipelines
429            .remove(&generate_pipeline_cache_key(schema, name, version));
430        self.pipelines
431            .remove(&generate_pipeline_cache_key(schema, name, None));
432        self.original_pipelines
433            .remove(&generate_pipeline_cache_key(schema, name, version));
434        self.original_pipelines
435            .remove(&generate_pipeline_cache_key(schema, name, None));
436
437        Ok(Some(()))
438    }

440    async fn find_pipeline(
441        &self,
442        schema: &str,
443        name: &str,
444        version: PipelineVersion,
445    ) -> Result<Option<(String, TimestampNanosecond)>> {
446        // 1. prepare dataframe
447        let dataframe = self
448            .query_engine
449            .read_table(self.table.clone())
450            .context(DataFrameSnafu)?;
451        let DataFrame::DataFusion(dataframe) = dataframe;
452
453        let dataframe = dataframe
454            .filter(prepare_dataframe_conditions(schema, name, version))
455            .context(BuildDfLogicalPlanSnafu)?
456            .select_columns(&[
457                PIPELINE_TABLE_PIPELINE_CONTENT_COLUMN_NAME,
458                PIPELINE_TABLE_CREATED_AT_COLUMN_NAME,
459            ])
460            .context(BuildDfLogicalPlanSnafu)?
461            .sort(vec![
462                col(PIPELINE_TABLE_CREATED_AT_COLUMN_NAME).sort(false, true)
463            ])
464            .context(BuildDfLogicalPlanSnafu)?
465            .limit(0, Some(1))
466            .context(BuildDfLogicalPlanSnafu)?;
467
468        let plan = dataframe.into_parts().1;
469
470        let table_info = self.table.table_info();
471
472        debug!("find_pipeline_by_name: plan: {:?}", plan);
473
474        // 2. execute plan
475        let output = self
476            .query_engine
477            .execute(plan, Self::query_ctx(&table_info))
478            .await
479            .context(ExecuteInternalStatementSnafu)?;
480        let stream = match output.data {
481            OutputData::Stream(stream) => stream,
482            OutputData::RecordBatches(record_batches) => record_batches.as_stream(),
483            _ => unreachable!(),
484        };
485
486        // 3. construct result
487        let records = record_util::collect(stream)
488            .await
489            .context(CollectRecordsSnafu)?;
490
491        if records.is_empty() {
492            return Ok(None);
493        }
494
495        // limit 1
496        ensure!(
497            records.len() == 1 && records[0].num_columns() == 2,
498            PipelineNotFoundSnafu { name, version }
499        );
500
501        let pipeline_content_column = records[0].column(0);
502        let pipeline_content = pipeline_content_column
503            .as_any()
504            .downcast_ref::<StringVector>()
505            .with_context(|| CastTypeSnafu {
506                msg: format!(
507                    "can't downcast {:?} array into string vector",
508                    pipeline_content_column.data_type()
509                ),
510            })?;
511
512        let pipeline_created_at_column = records[0].column(1);
513        let pipeline_created_at = pipeline_created_at_column
514            .as_any()
515            .downcast_ref::<TimestampNanosecondVector>()
516            .with_context(|| CastTypeSnafu {
517                msg: format!(
518                    "can't downcast {:?} array into scalar vector",
519                    pipeline_created_at_column.data_type()
520                ),
521            })?;
522
523        debug!(
524            "find_pipeline_by_name: pipeline_content: {:?}, pipeline_created_at: {:?}",
525            pipeline_content, pipeline_created_at
526        );
527
528        ensure!(
529            pipeline_content.len() == 1,
530            PipelineNotFoundSnafu { name, version }
531        );
532
533        // Safety: asserted above
534        Ok(Some((
535            pipeline_content.get_data(0).unwrap().to_string(),
536            pipeline_created_at.get_data(0).unwrap(),
537        )))
538    }
